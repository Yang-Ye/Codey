class Solution {
public:
    int jump(int A[], int n) {
        int curMax = 0, njumps = 0, i = 0;
        while(curMax<n-1) {
            int lastMax = curMax;
            for(; i<=lastMax; i++) 
                curMax = max(curMax,i+A[i]);
            njumps++;
            if(lastMax == curMax) return -1;
        }
        return njumps;
    }
};
