class Solution(object):
    def reversePairs(self, nums):
        return self.helper(nums, 0, len(nums))
        
    def helper(self, nums, l, r):
        mid = l + r >> 1
        if mid == l: return 0
        total = self.helper(nums, l, mid) + self.helper(nums, mid, r)
        prev_total = 0

            prev_total += idx - mid
            mid = idx
            total += prev_total
        nums[l: r] = sorted(nums[l: r])
        return total
