class Solution(object):
    def findMaxConsecutiveOnes(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if len(nums) == sum(nums): return sum(nums)
        a = b = c = 0
        for n in nums:
            if n == 1:
                c += 1
            else:
                b, c = c, 0
            a = max(a, b + c + 1)
        return a
