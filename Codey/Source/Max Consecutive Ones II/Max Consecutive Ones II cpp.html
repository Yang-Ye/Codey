class Solution {
public:
    int findMaxConsecutiveOnes(vector<int>& nums) {
        int start=0, usedup=0, maxl=0;
        for (int i=0; i<nums.size(); ++i) {
            if (nums[i]==0) {
                if (!usedup) usedup=1;
                else {
                    while(nums[start] !=0) start++;
                    ++start;
                }
            }
            maxl = max(maxl, i-start+1);
            
        }
        return maxl;
        
    }
};
