class Solution {  
public:  
    bool isConvex(vector<vector<int>>& points) {  
        long prev = 0;  
        int n = points.size();  
        for(int i = 0; i < n; i++) {  
            vector<vector<int>> m;  
            for(int j = 1; j < 3; j++) {  
                m.push_back({  
                    points[(i+j)%n][0] - points[i][0],  
                    points[(i+j)%n][1] - points[i][1]  
                });  
            }  
            long curr = det(m);  
            if (curr * prev < 0) return false;  
            prev = curr;  
        }  
        return true;  
    }  
    long det(vector<vector<int>>& m) {  
        return m[0][0] * m[1][1] - m[0][1] * m[1][0];  
    }  
};
