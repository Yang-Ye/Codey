public static int[] constructRectangle(int area) {
        int[] result = new int[2];
        if(area == 0){
            return result;
        }
        int length = (int)(Math.ceil(Math.sqrt(area)));
        while(area%length != 0){
            length++;
        }
        int width = area/length;
        result[0] = length;
        result[1] = width;
        return result;
    }
