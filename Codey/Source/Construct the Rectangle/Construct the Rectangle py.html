class Solution(object):
    def constructRectangle(self, area):
        """
        :type area: int
        :rtype: List[int]
        """
        # Make w start from int(sqrt(area)) and reduce to 1, stop if area % w == 0
        from math import sqrt
        w = int(sqrt(area))
        while w > 1:
            if area % w == 0:
                return [area / w, w]
            w -= 1
        return [area, 1]
